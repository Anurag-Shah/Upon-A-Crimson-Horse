# Rebellion in Isfahan
country_event = {
	id = UaCH.8 
	title = UaCH.8.t 
	desc = UaCH.8.d
	picture = BORDER_TENSION_eventPicture
	
	fire_only_once = yes
	trigger = {
		AND = {
			tag = ISF
			has_country_flag = isfh_rebels
		}
		NOT = { has_global_flag = uach_succession_crisis_started }
		NOT = { TIM = { has_country_flag = uach_interwar_lull } }
	}
	mean_time_to_happen = {
		months = 10
	}
	major = yes
	major_trigger = {
		is_governor_or_prince = yes
	}

	# Send our demands to the governors!
	option = {
		name = "UaCH.8.a"

		if = {
			limit = {
				is_subject_of = QOM
			}
			QOM = {
				declare_war_with_cb = {
					casus_belli = cb_rebellion_of_sultan_muhammad
					who = TIM
				}
			}
		}
		else = {
			declare_war_with_cb = {
				casus_belli = cb_rebellion_of_sultan_muhammad
				who = TIM
			}
		}	

		custom_tooltip = uach_send_demands
		hidden_effect = {
			every_country = {
				limit = {
					is_governor_or_prince = yes
					NOT = { tag = TIM }
					NOT = {
						AND = {
							tag = QOM 
							ISF = {
								is_subject_of = QOM
							}
						} 
					}
					NOT = { tag = ISF }
					NOT = { tag = QOM }
				}
				country_event = {
					id = UaCH.9
					days = 1
				}
			}
			TIM = {
				# Inform TIM about the rebellion
				country_event = {
					id = UaCH.108
				}
			}
		}

		ai_chance = {
			factor = 1000
			modifier = {
				factor = 100
				always = yes
			}
		}
	}

	# Expel him, lest you bring Shah Rukhs wrath on our head!
	option = {
		name = "UaCH.8.b"
		
		add_stability = -3
		add_legitimacy = -100
		
		if = {
			limit = {
				OR = {
					has_country_modifier = mnt_isf_liberty_desire
					has_country_modifier = mnt_isf_liberty_desire_60
				}
			}
			remove_country_modifier = mnt_isf_liberty_desire
			remove_country_modifier = mnt_isf_liberty_desire_60
		}

		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
			}
		}
	}
}
