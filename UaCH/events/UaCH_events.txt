# Rebellion in Isfahan
country_event = {
	id = UaCH.8 
	title = UaCH.8.t 
	desc = UaCH.8.d
	picture = CITY_DEVELOPMENT_eventPicture
	
	fire_only_once = yes
	trigger = {
		AND = {
			tag = ISF
			has_country_flag = isfh_rebels
		}
	}
	mean_time_to_happen = {
		months = 30
	}
	major = yes
	major_trigger = {
		is_governor_or_prince = yes
	}

	# Send our demands to the governors!
	option = {
		name = "UaCH.8.a"

		if = {
			limit = {
				is_subject_of = QOM
			}
			QOM = {
				declare_war_with_cb = {
					casus_belli = cb_rebellion_of_sultan_muhammad
					who = TIM
				}
			}
		}
		else = {
			declare_war_with_cb = {
				casus_belli = cb_rebellion_of_sultan_muhammad
				who = TIM
			}
		}	

		custom_tooltip = uach_send_demands
		hidden_effect = {
			every_country = {
				limit = {
					is_governor_or_prince = yes
					NOT = { tag = TIM }
					NOT = {
						AND = {
							tag = QOM 
							ISF = {
								is_subject_of = QOM
							}
						} 
					}
				}
				country_event = {
					id = UaCH.9
					days = 1
				}
			}
		}

		ai_chance = {
			factor = 1000
			modifier = {
				factor = 100
				always = yes
			}
		}
	}

	# Expel him, lest you bring Shah Rukhs wrath on our head!
	option = {
		name = "UaCH.8.b"
		
		add_stability = -3
		add_legitimacy = -100
		
		if = {
			limit = {
				OR = {
					has_country_modifier = mnt_isf_liberty_desire
					has_country_modifier = mnt_isf_liberty_desire_60
				}
			}
			remove_country_modifier = mnt_isf_liberty_desire
			remove_country_modifier = mnt_isf_liberty_desire_60
		}

		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
			}
		}
	}
}

# OPM country choose independence or choose to side with someone
# They will side with a random country that:
#	A. doesnt have a core on their province
#	B. borders them
#	C. isn't an OPM
#	D. Is a prince / governor
# Only ever fires for AI
# So DW about localizing this
country_event = {
	id = UaCH.64
	title = UaCH.64.t 
	desc = UaCH.64.d
	picture = BIG_ARABIAN_CITY_eventPicture
	is_triggered_only = yes

	# Swear Fealty to a neighbor
	option = {
		name = "UaCH.64.a"
		trigger = {
			any_neighbor_country = {
				AND = {
					NOT = { core_claim = ROOT }
					num_of_provinces_in_states = 2
					OR = {
						is_prince = yes
						is_governor = yes
					}
				}
			}
		}

		clr_country_flag = uach_still_considering

		# This has to be the dumbest workaround of all time
		# paradox......
		if = {
			limit = {
				AND = {
					NOT = { core_claim = ROOT }
					num_of_provinces_in_states = 2
					OR = {
						is_prince = yes
						is_governor = yes
					}
				}
			}
			random_neighbor_country = {
				limit = {
					AND = {
						NOT = { core_claim = ROOT }
						num_of_provinces_in_states = 2
						OR = {
							is_prince = yes
							is_governor = yes
						}
					}
				}

				create_subject = {
					subject_type = governor
					who = PREV
				}
			}
		} else = {
			uach_add_governor_to_war = yes
		}

		ai_chance = { factor = 80 }
	}

	# Join the war
	option = {
		name = "UaCH.64.b"
		uach_add_governor_to_war = yes
	}
}
