# Rebellion in Isfahan
country_event = {
	id = UaCH.8 
	title = UaCH.8.t 
	desc = UaCH.8.d
	picture = CITY_DEVELOPMENT_eventPicture
	
	fire_only_once = yes
	trigger = {
		AND = {
			tag = ISF
			has_country_flag = isfh_rebels
		}
	}
	mean_time_to_happen = {
		months = 30
	}
	major = yes
	major_trigger = {
		is_governor_or_prince = yes
	}

	# Send our demands to the governors!
	option = {
		name = "UaCH.8.a"

		if = {
			limit = {
				is_subject_of = QOM
			}
			QOM = {
				declare_war_with_cb = {
					casus_belli = cb_rebellion_of_sultan_muhammad
					who = TIM
				}
			}
		}
		else = {
			declare_war_with_cb = {
				casus_belli = cb_rebellion_of_sultan_muhammad
				who = TIM
			}
		}	

		custom_tooltip = uach_send_demands
		hidden_effect = {
			every_country = {
				limit = {
					is_governor_or_prince = yes
					NOT = { tag = TIM }
					NOT = {
						AND = {
							tag = QOM 
							ISF = {
								is_subject_of = QOM
							}
						} 
					}
				}
				country_event = {
					id = UaCH.9
					days = 1
				}
			}
		}

		ai_chance = {
			factor = 1000
			modifier = {
				factor = 100
				always = yes
			}
		}
	}

	# Expel him, lest you bring Shah Rukhs wrath on our head!
	option = {
		name = "UaCH.8.b"
		
		add_stability = -3
		add_legitimacy = -100
		
		if = {
			limit = {
				OR = {
					has_country_modifier = mnt_isf_liberty_desire
					has_country_modifier = mnt_isf_liberty_desire_60
				}
			}
			remove_country_modifier = mnt_isf_liberty_desire
			remove_country_modifier = mnt_isf_liberty_desire_60
		}

		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
			}
		}
	}
}

			# If FROM is an interloper, restart the war (we know ROOT isnt an interloper)
			else_if = {
				limit = {
					FROM = {
						is_interloper = yes
					}
				}

				FROM = {
					declare_war_with_cb	= {
						casus_belli = cb_uach_interloper_conquest
						who = ROOT
					}
					country_event = { id = UaCH.74 }
				}
				ROOT = { country_event = { id = UaCH.75 }}
			}

			# Neither country is an interloper
			# If both countries are governors, ~1 year ceasefire
			else_if = {
				limit = {
					FROM = {
						is_governor = yes
					}
					ROOT = {
						is_governor = yes
					}
				}

				FROM = {
					# Restart war in average 1 year, between 300-430 days
					country_event = {
						id = UaCH.70
						days = 300
						random = 130
					}
				}

				# Inform both sides about the ceasefire
				FROM = {
					country_event = {
						id = UaCH.73
					}
				}
			}

			# At least 1 country is a prince
			# Declare war in ~6 months

			else = {
				FROM = {
					# Restart war in average 6 months, between 150-215 days
					country_event = {
						id = UaCH.70
						days = 150
						random = 65
					}
				}

				# Inform both sides about the ceasefire
				FROM = {
					country_event = {
						id = UaCH.73
					}
				}
			}
		}
	}
}
