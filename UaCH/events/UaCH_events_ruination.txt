#event start
country_event = {
	id = UaCH.escalation.1 
	title = UaCH.escalation.1.t 
	desc = UaCH.escalation.1.d
	picture = CITY_DEVELOPMENT_eventPicture
	is_triggered_only = yes
	
	trigger = { #disaster start event calls}

	hidden = yes
	
	option = {
		name = "UaCH.1.escalation.a"
		hidden_effect = {
			every_country = {
				limit = {
					has_government_reform = uach_governor		
				}
				add_country_modifier ={ 
					name = uach_escalation_governor_1
					duration = -1 }
			}
			every_country = {
				limit = {
					AND  = {
						has_government_reform = uach_prince
						is_timurid_emperor = no
					}
				}
				add_country_modifier ={  
					name = uach_escalation_prince_1
					duration = -1 }
			}
			every_country = {
				limit = {
					AND  = {
						has_government_reform = uach_prince
						is_timurid_emperor = yes
					}
				
				}
				add_country_modifier = { 
					name = uach_escalation_emperor_1
					duration = -1}
			}
			
			set_global_flag = uach_escalation_timer
		
			}
	}
}

country_event = {
	id = UaCH.escalation.3
	title = UaCH.escalation.3.t 
	desc = UaCH.escalation.3.d
	picture = CITY_DEVELOPMENT_eventPicture
	is_triggered_only = yes
	
	trigger = { #disaster end event calls}

	hidden = yes
	
	option = {
		name = "UaCH.3.escalation.a"
		hidden_effect = {
			every_country = {
				limit = {
					has_government_reform = uach_governor		
				}
				remove_country_modifier = uach_escalation_governor_1
				remove_country_modifier = uach_escalation_governor_2
				remove_country_modifier = uach_escalation_governor_3
				remove_country_modifier = uach_escalation_governor_4
				remove_country_modifier = uach_escalation_governor_5
				remove_country_modifier = uach_escalation_governor_6

			}
			every_country = {
				limit = {
					AND  = {
						has_government_reform = uach_prince
						is_timurid_emperor = no
					}
				}
				remove_country_modifier = uach_escalation_prince_1
				remove_country_modifier = uach_escalation_prince_2
				remove_country_modifier = uach_escalation_prince_3
				remove_country_modifier = uach_escalation_prince_4
				remove_country_modifier = uach_escalation_prince_5
				remove_country_modifier = uach_escalation_prince_6
			}
			every_country = {
				limit = {
					AND  = {
						has_government_reform = uach_prince
						is_timurid_emperor = yes
					}
				
				}
				IF =  { has_country_modifier = uach_escalation_emperor_1 {
						  	remove_country_modifier = uach_escalation_emperor_1
							add_country_modifier = uach_escalation_rew_lvl1
							duration = -1}
						}	
				IF =  { has_country_modifier = uach_escalation_emperor_2 {
						  	remove_country_modifier = uach_escalation_emperor_2
							add_country_modifier = uach_escalation_rew_lvl2
							duration = -1}
						}	
				IF =  { has_country_modifier = uach_escalation_emperor_3 {
						  	remove_country_modifier = uach_escalation_emperor_3
							add_country_modifier = uach_escalation_rew_lvl3
							duration = -1}
						}	
				IF =  { has_country_modifier = uach_escalation_emperor_4 {
						  	remove_country_modifier = uach_escalation_emperor_4
							add_country_modifier = uach_escalation_rew_lvl4
							duration = -1}
						}	
				IF =  { has_country_modifier = uach_escalation_emperor_5 {
						  	remove_country_modifier = uach_escalation_emperor_5
							add_country_modifier = uach_escalation_rew_lvl5
							duration = -1}
						}	
						
				IF =  { has_country_modifier = uach_escalation_emperor_6 {
						  	remove_country_modifier = uach_escalation_emperor_6
							add_country_modifier = uach_escalation_rew_lvl6
							duration = -1}
						}					

			}
			
			clear_global_flag = uach_escalation_timer
		}
	}
}
