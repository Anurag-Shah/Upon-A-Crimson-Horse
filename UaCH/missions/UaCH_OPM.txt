opm_missions_slot3 = {
	slot = 3
	generic = no
	ai = yes
	has_country_shield = yes
	potential = {
		is_governor = yes
		NOT = {
			tag = LRE
			tag = AFG
			tag = ISF
			tag = KHI
			tag = SIS
			tag = KRM
			tag = KHO
			tag = ARD
			tag = KHU
			tag = FRS
			tag = LRI
		}
	}

	uach_opm_31 = {
		icon = uach_opm_31
		position = 1
		provinces_to_highlight = {  } 

		trigger = { num_of_generals = 2 } 

		effect = {
			random_owned_province = {
				add_unit_construction = {
					type = cavalry
					cost = 0
					speed = -0.25
					amount = 3
				}
				add_unit_construction = {
					type = infantry
					cost = 0
					speed = -0.25
					amount = 3
				}
			}
		}
	}

	uach_opm_32 = {
		icon = uach_opm_32
		position = 2
		required_missions = { uach_opm_31 } 
		provinces_to_highlight = {  } 

		trigger = { any_owned_province = { fort_level = 3 }  } 

		effect = {
			add_legitimacy = 50
			add_adm_power = 50
		}
	}
}
