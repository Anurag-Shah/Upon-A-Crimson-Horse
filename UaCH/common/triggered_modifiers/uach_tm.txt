uach_reform_at_peace = {
    potential = {
        AND = {
            tag = TIM
            has_country_flag = uach_interwar_lull
        }
    }
    trigger = {
        NOT = { is_at_war = yes }
    }

    monthly_uach_persianization = 0.2
}

uach_difficulty_medium = {
	potential = {
		is_governor_or_prince = yes
		ai = yes
	}

	trigger = {
		AND = {has_global_flag = uach_diff_medium
			OR = {
					has_disaster = timurid_succession_crisis
					is_in_war = { casus_belli = cb_challange_claimant }
					is_in_war = { casus_belli = cb_enforce_loyalty_governor }
				}
				OR = { 
				ai = yes
				if = {{is_subject = yes}
						overlord = { ai = yes }}
				}
			}
		}
	land_maintenance_modifier = -0.10
	manpower_recovery_speed = 0.1
	global_tax_modifier = 0.15
	
}
