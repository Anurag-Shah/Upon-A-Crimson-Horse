# This is mechanics for prince

uach_mechanics_prince = {
	alert_icon_gfx = GFX_alerticons_government_mechanics
	alert_icon_index = 18

	available = {
		NOT = { has_country_flag = uach_interwar_lull }
	}
	
	interactions = {
		# Raise a tax: gain income at the cost of base unrest
		uach_raise_a_tax = {
			icon = UACH_GFX_tax_button
			cooldown_years = 1
			cooldown_token = iqta_mechanic
			cooldown_desc = UACH_GOV_REFORM_COOLDOWN

			trigger = {
				custom_trigger_tooltip = {
					tooltip = WAIT_FOR_SC_TO_START_BITCH_TT
					has_global_flag = uach_succession_crisis_started
				}
			}
			effect = {
				add_years_of_income = 1.5
				custom_tooltip = uach_conscript_men_effect_tt
				hidden_effect = {
					every_province = {
						limit = {
							controlled_by = PREV
						}
						add_unrest = 3
					}
				}
			}
			ai_chance = {
				# This is the default choice for the AI, factor will never be 0
				# This is because we like rebels!
				# Rebels are free troops that add random chaos to the civil war
				# This is a good thing
				# Because I am evil
				factor = 10
				modifier = {
					factor = 0.2
					is_in_deficit = no
				}
				modifier = {
					factor = 0.1
					NOT = { num_of_loans = 1 }
				}
				modifier = {
					factor = 2
					num_of_loans = 5
				}
				modifier = {
					factor = 4
					num_of_loans = 10
				}
				modifier = {
					factor = 6
					num_of_loans = 15
				}
			}
		}



		# Request foreign support
		uach_request_foreign_support = {
			icon = UACH_GFX_foreign_support_button
			cooldown_years = 1
			cooldown_token = iqta_mechanic
			cooldown_desc = UACH_GOV_REFORM_COOLDOWN
			trigger = {
				custom_trigger_tooltip = {
					tooltip = UACH_GOV_RFS_TOOLTIP
					calc_true_if = {
						all_neighbor_country = {
							NOT = {
								is_governor = yes
								is_prince = yes
								is_interloper = yes
							}
						}
						amount = 1
					}
				}
				custom_trigger_tooltip = {
					tooltip = WAIT_FOR_SC_TO_START_BITCH_TT
					has_global_flag = uach_succession_crisis_started
				}
			}
			effect = {
				add_yearly_manpower = 2
				add_years_of_income = 2
				add_legitimacy = -10
				add_prestige = -15
				random_neighbor_country = {
					add_opinion = {
						who = PREV
						modifier = supported_us_in_timurid_success_crisis
					}
					add_prestige = 5
				}
			}
			ai_chance = {
				# The AI will never do this if they have no loans / a bank of manpower (10% max)
				# Then if they have both loans and manpower, the factor will increase
				# They are unlikely to do it without a defecit
				factor = 10
				modifier = {
					factor = 0
					NOT = {
						num_of_loans = 1
						manpower_percentage = 0.1
					}
				}
				modifier = {
					factor = 0.5
					is_in_deficit = no
				}
				modifier = {
					factor = 2
					num_of_loans = 5
				}
				modifier = {
					factor = 4
					num_of_loans = 10
				}
				modifier = {
					factor = 3
					manpower_percentage = 0.05
				}
				modifier = {
					factor = 4
					manpower_percentage = 0.01
				}
			}
		}



		
		# Raise support for claimant
		uach_raise_claimant_support = {
			icon = UACH_GFX_strengthen_claim_button
			cooldown_years = 1
			cooldown_token = iqta_mechanic
			cooldown_desc = UACH_GOV_REFORM_COOLDOWN
			trigger = {
				custom_trigger_tooltip = {
					tooltip = UACH_GOV_RCS_TT
					AND = {
						is_prince = yes
						# at least 1 unoccupied province not under seige
						any_province = {
							AND = {
								controlled_by = PREV
								NOT = {
									has_siege = yes
								}
							}
						}
					}
				}
				custom_trigger_tooltip = {
					tooltip = WAIT_FOR_SC_TO_START_BITCH_TT
					has_global_flag = uach_succession_crisis_started
				}
			}
			effect = {
				add_war_exhaustion = -3
				add_legitimacy = 5
				add_years_of_income = -0.25
				add_prestige = -10
				add_adm_power = -25
				add_dip_power = -25
				add_mil_power = -25
				if = {
					limit = {
						capital_scope = {
							AND = {
								controlled_by = PREV
								NOT = {
									has_siege = yes
								}
							}
						}
					}
					capital_scope = {
						add_unit_construction = {
							type = infantry
							amount = 3
							speed = 0.25
							cost = 0
						}
						add_unit_construction = {
							type = cavalry
							amount = 3
							speed = 0.25
							cost = 0
						}
					}
				}
			}
			ai_chance = {
				# The AI will never do this if they have no loans / a bank of manpower (10% max)
				# Then if they have both loans and manpower, the factor will increase
				# They are unlikely to do it without a defecit
				factor = 10
				modifier = {
					factor = 0
					NOT = {
						num_of_loans = 1
						manpower_percentage = 0.1
					}
				}
				modifier = {
					factor = 0.5
					is_in_deficit = no
				}
				modifier = {
					factor = 2
					num_of_loans = 5
				}
				modifier = {
					factor = 4
					num_of_loans = 10
				}
				modifier = {
					factor = 3
					manpower_percentage = 0.05
				}
				modifier = {
					factor = 4
					manpower_percentage = 0.01
				}
			}
		}
	}
}

uach_reform_state_mechanic = {
	alert_icon_gfx = GFX_alerticons_government_mechanics
	alert_icon_index = 18
	available = {
		has_country_flag = uach_interwar_lull
		tag = TIM
	}

	powers = {
		uach_persianization = {
			min = -100
			max = 100
			default = 0
			show_before_interactions = yes

			scaled_modifier = {
				start_value = 0
				end_value = 100
				extend_beyond_value = no
				trigger = {}
				modifier = {
					reduced_liberty_desire = -30
					global_tax_modifier = 0.2
					global_unrest = -3
				}
			}

			reverse_scaled_modifier = {
				start_value = -100
				end_value = 0
				extend_beyond_value = no
				trigger = {}
				modifier = {
					reduced_liberty_desire = -40
					morale = 10
				}
			}

			range_modifier = {
				start_value = -100
				end_value = -66
				extend_beyond_value = no
				trigger = {}
				modifier = {
					raze_province = yes
					raze_power_gain = 0.1
				}
			}
		}
	}

	interactions = {
		uach_fully_mongolize = {
			icon = UACH_GFX_strengthen_claim_button

			trigger = {
				NOT = {
					has_government_power = {
						mechanic_type = uach_reform_state_mechanic
						power_type = uach_persianization
						value = -99.9
					}
				}
			}
			effect = {
				custom_tooltip = UACH_P2FM_R_TT
				hidden_effect = {
					set_country_flag = timurids_reformed_government
					every_subject_country = {
						NOT = { liberty_desire = 50 }
						PREV = {
							inherit = PREV
						}
					}
					# Vassalize all disloyal subjects
					every_subject_country = {
						limit = {
							OR = {
								is_prince = yes
								is_governor = yes
							}
						}
						grant_independence = yes
						PREV = {
							vassalize = PREV
						}
					}
				}
				remove_government_reform = uach_prince
				change_government = tribal
				add_government_reform = uach_heirs_of_timur_reform
				regenerate_government_mechanics = yes
				add_country_modifier = {
					name = "centralization_modifier"
					duration = 7300
				}
				hidden_effect = {
					country_event = { id = ideagroups.1 }
					clr_country_flag = uach_interwar_lull
				}
			}
			ai_chance = { # AI will do this if it can
				factor = 100
			}
		}

		# Request foreign support
		uach_scrap_a_reform = {
			icon = UACH_GFX_foreign_support_button
			cooldown_years = 2
			cooldown_token = iqta_mechanic
			cooldown_desc = UACH_SCRAP_REFORM_COOLDOWN_TT
			trigger = {
				OR = {
					has_government_power = {
						mechanic_type = uach_reform_state_mechanic
						power_type = uach_persianization
						value = 10
					}
					NOT = {
						has_government_power = {
							mechanic_type = uach_reform_state_mechanic
							power_type = uach_persianization
							value = -10
						}
					}
				}
			}
			effect = {
				uach_add_or_subtract_persianization = { value = 10 }
				every_subject_country = {
					add_liberty_desire = -10
				}
			}
			ai_chance = {
				# The AI will never do this if they have no loans / a bank of manpower (10% max)
				# Then if they have both loans and manpower, the factor will increase
				# They are unlikely to do it without a defecit
				factor = 10
				modifier = {
					factor = 0
					NOT = {
						any_subject_country = {
							liberty_desire = 50
						}
					}
				}
				modifier = {
					factor = 1.2
					any_subject_country = {
						liberty_desire = 75
					}
				}
				modifier = {
					factor = 1.3
					any_subject_country = {
						liberty_desire = 100
					}
				}
				modifier = {
					factor = 2
					calc_true_if = {
						all_subject_country = {
							liberty_desire = 50
						}
						amount = 3
					}
				}
				modifier = {
					factor = 20
					calc_true_if = {
						all_subject_country = {
							liberty_desire = 50
						}
						amount = 5
					}
				}
			}
		}

		uach_fully_persianize = {
			icon = UACH_GFX_tax_button

			trigger = {
				has_government_power = {
					mechanic_type = uach_reform_state_mechanic
					power_type = uach_persianization
					value = 100
				}
			}
			effect = {
				# Base game
				change_tag = PER
				on_change_tag_effect = yes
				swap_non_generic_missions = yes
				remove_non_electors_emperors_from_empire_effect = yes
				if = {
					limit = {
						NOT = { government = monarchy }
						is_revolutionary = no
					}
					change_government = monarchy
				}
				set_government_rank = 3
				add_country_modifier = {
					name = uach_persianized
					duration = 3650
				}
				add_prestige = 25
				persia_region = {
					limit = {
						NOT = { owned_by = ROOT }
						NOT = { is_core = ROOT }
					}
					add_permanent_claim = PER
				}
				add_country_modifier = {
					name = "centralization_modifier"
					duration = 7300
				}
				custom_tooltip = UACH_PERSIANIZED_EFFECT_RESULT_TT
				hidden_effect = {
					set_country_flag = timurids_reformed_government
					clr_country_flag = uach_interwar_lull
					# Integrate all loyal subjects
					every_subject_country = {
						NOT = { liberty_desire = 50 }
						PREV = {
							inherit = PREV
						}
					}
					# Vassalize all disloyal subjects
					every_subject_country = {
						limit = {
							OR = {
								is_prince = yes
								is_governor = yes
							}
						}
						grant_independence = yes
						PREV = {
							vassalize = PREV
						}
					}
					if = {
						limit = { is_revolutionary = no NOT = { religion_group = zoroastrian_group } }
						add_government_reform = persian_government_reform
					}
					if = {
						limit = { is_revolutionary = no religion_group = zoroastrian_group }
						add_government_reform = zoroastrian_persian_government_reform
					}
					if = {
						limit = {
							has_reform = states_general_reform
						}
						remove_government_reform = states_general_reform
					}
				}
				if = {
					limit = {
						OR = {
							NOT = { has_dlc = "King of Kings" }
							NOT = { primary_culture = mazandarani }
						}
						NOT = { primary_culture = persian }
					}
					change_primary_culture = persian
				}
				set_country_flag = formed_persia_flag
				country_event = { id = ideagroups.1 }
				if = {
					limit = { has_dlc = "King of Kings" }
					unlock_estate = {
						estate = estate_qizilbash
					}
				}
			}
			ai_chance = { # AI will do this if it can
				factor = 100
			}
		}
	}
